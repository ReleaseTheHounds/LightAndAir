uusing System;
using System.Linq;
using System.Windows.Forms;

namespace WinFormsEFComboBox
{
    public partial class MainForm : Form
    {
        private AppDbContext _context;

        public MainForm()
        {
            InitializeComponent();
            _context = new AppDbContext();
            LoadCategories();
        }

        // Load data into ComboBox
        private void LoadCategories()
        {
            comboBoxCategories.DataSource = _context.Categories.ToList();
            comboBoxCategories.DisplayMember = "Name";
            comboBoxCategories.ValueMember = "Id";
        }

        // CREATE
        private void btnAdd_Click(object sender, EventArgs e)
        {
            string name = txtCategoryName.Text.Trim();
            if (string.IsNullOrEmpty(name))
            {
                MessageBox.Show("Please enter a category name.");
                return;
            }

            _context.Categories.Add(new Category { Name = name });
            _context.SaveChanges();
            LoadCategories();
            txtCategoryName.Clear();
        }

        // UPDATE
        private void btnUpdate_Click(object sender, EventArgs e)
        {
            if (comboBoxCategories.SelectedValue == null) return;

            int id = (int)comboBoxCategories.SelectedValue;
            var category = _context.Categories.Find(id);
            if (category != null)
            {
                category.Name = txtCategoryName.Text.Trim();
                _context.SaveChanges();
                LoadCategories();
            }
        }

        // DELETE
        private void btnDelete_Click(object sender, EventArgs e)
        {
            if (comboBoxCategories.SelectedValue == null) return;

            int id = (int)comboBoxCategories.SelectedValue;
            var category = _context.Categories.Find(id);
            if (category != null)
            {
                _context.Categories.Remove(category);
                _context.SaveChanges();
                LoadCategories();
            }
        }

        // Populate textbox when selection changes
        private void comboBoxCategories_SelectedIndexChanged(object sender, EventArgs e)
        {
            if (comboBoxCategories.SelectedItem is Category selected)
            {
                txtCategoryName.Text = selected.Name;
            }
        }
    }
}
